def round_df(df: pd.DataFrame, decimals=3) -> pd.DataFrame:
    """Rounds all numeric values in a DataFrame up to a specified number of decimals."""
    factor = 10 ** decimals
    return df.apply(lambda col: np.ceil(col * factor) / factor if np.issubdtype(col.dtype, np.number) else col, axis=0)

class HydrologicSimulator:
    """
    Encapsulates the data and methods for a watershed hydrologic simulation.
    """
    def __init__(self, graph: nx.DiGraph, outdir: str,
                 start_time: float = 0.0, end_time: float = 72.0, time_interval: float = 1.0,
                 column_mapping: dict = None, conversion_factor: float = 0.0000229568):
        """
        Initializes the simulator with all necessary configuration.

        Parameters:
        - graph (nx.DiGraph): NetworkX graph with node attributes.
        - outdir (str): Output directory for result files.
        - start_time (float): The simulation start time in hours. Defaults to 0.
        - end_time (float): The simulation end time in hours. Defaults to 72.
        - time_interval (float): The time step for the simulation in hours. Defaults to 1.
        - column_mapping (dict, optional): Overrides for default column names.
        - conversion_factor (float, optional): To convert cfs to acre-ft/hr.
        """
        self.graph = graph
        self.outdir = outdir
        self.start_time = start_time
        self.end_time = end_time
        self.time_interval = time_interval
        # Generate the list of time steps for the routing loop
        self.time_steps = np.arange(start_time, end_time + time_interval, time_interval)
        
        self.outlet_tables = {}
        self.conversion_factor = conversion_factor

        # Define default parameters and update with any user-provided overrides
        self.params = {
            'rainfall_col': '1d 100y Rainfall (in)', 'soil_storage_col': 'Soil Sorage (in)',
            'tc_col': 'Concentration (hr)', 'area_col': 'Area (ac)',
            'node_type_col': 'Node Type', 'initial_stage_col': 'Initial Stage (ft)',
            'high_elev_col': 'High Elevation (ft)'
        }
        if column_mapping:
            self.params.update(column_mapping)

    def run(self):
        """
        Executes the entire hydrologic simulation from start to finish.
        """
        print("🚀 Starting hydrologic simulation...")
        sorted_nodes = list(nx.topological_sort(self.graph))

        for node in sorted_nodes:
            attrs = self.graph.nodes[node]
            if str(attrs.get(self.params['node_type_col'], '')).strip().lower() != "outlet":
                continue

            required_cols = [
                self.params['area_col'], self.params['tc_col'], self.params['rainfall_col'],
                self.params['soil_storage_col'], self.params['initial_stage_col'], self.params['high_elev_col']
            ]
            if any(attrs.get(c) is None for c in required_cols):
                print(f"⚠️  Skipping node '{attrs.get('label', node)}' due to missing critical attributes.")
                continue

            print(f"Processing outlet: {attrs.get('label', node)}...")
            inflow_data = self._generate_inflow_hydrograph(attrs)
            results_df = self._route_hydrograph(node, inflow_data)
            self.outlet_tables[attrs.get("label", node)] = results_df

        self._save_results()

    def _route_hydrograph(self, node: str, inflow_data: pd.DataFrame) -> pd.DataFrame:
        """Routes the inflow hydrograph through an outlet node."""
        attrs = self.graph.nodes[node]
        min_elev, max_elev, A = attrs.get(self.params['initial_stage_col']), attrs.get(self.params['high_elev_col']), attrs.get(self.params['area_col'])
        downstream_nodes = list(self.graph.successors(node))

        df_out, state = self._initialize_routing(node, downstream_nodes)

        for t in self.time_steps:
            current_inflow = inflow_data.loc[t] if t in inflow_data.index else pd.Series(0.0, index=inflow_data.columns)
            runoff_cfs = current_inflow.get('Runoff (cfs)', 0.0)
            runoff_af = np.round(runoff_cfs * 3600 * self.conversion_factor, 3)
            state['cumulative_runoff'] += runoff_af

            total_q, discharge_per_node = self._calculate_total_discharge(state['stage'], downstream_nodes)
            total_q_af = np.round(total_q * 3600 * self.conversion_factor, 3)
            state['cumulative_total_discharge'] += total_q_af

            state['storage'] = max(0.0, state['storage'] + runoff_af - total_q_af)
            state['stage'] = simple_average(state['storage'], min_elev, max_elev, A)

            self._record_timestep_results(df_out, t, current_inflow, state, runoff_cfs, runoff_af, total_q, discharge_per_node)
            
        return pd.DataFrame(df_out)
        
    def _initialize_routing(self, node: str, downstream_nodes: list):
        """Initializes data structures for the routing process."""
        df_out = {
            "Time (hr)": [], "Cumulative Rainfall (in)": [], "Incremental Rainfall (in)": [],
            "Rainfall Excess (in)": [], "Runoff (cfs)": [], "Runoff (acre-ft)": [],
            "Cumulative Runoff (acre-ft)": [], "Stage (ft NGVD)": [], "Storage (acre-ft)": [],
            "Total Discharge (cfs)": [], "Cumulative Total Discharge (acre-ft)": [],
            **{f"{self.graph.nodes[dn].get('label', dn)} Discharge (cfs)": [] for dn in downstream_nodes}
        }
        state = {
            'storage': 0.0, 'stage': self.graph.nodes[node].get(self.params['initial_stage_col']), 
            'cumulative_runoff': 0.0, 'cumulative_total_discharge': 0.0
        }
        return df_out, state
        
    def _calculate_total_discharge(self, stage: float, downstream_nodes: list) -> tuple[float, dict]:
        """Calculates total discharge from all downstream structures."""
        total_q = 0.0
        discharge_per_node = {}
        for dn in downstream_nodes:
            dn_attrs = self.graph.nodes[dn]
            dn_type = str(dn_attrs.get(self.params['node_type_col'], '')).lower()
            label = dn_attrs.get('label', dn)
            q = 0.0
            
            on_elev = dn_attrs.get(self.params.get('on_elev_col', 'On Elev'), 5.0)
            capacity = dn_attrs.get(self.params.get('capacity_col', 'Capacity(cfs)'), 3240.22)
            crest_elev = dn_attrs.get(self.params.get('crest_elev_col', 'Crest Elev'), 4.5)
            width = dn_attrs.get(self.params.get('width_col', 'Width'), 10.0)
            design_head = dn_attrs.get(self.params.get('design_head_col', 'Design Head'), 2.0)
            num_piers = dn_attrs.get(self.params.get('num_piers_col', 'Num Piers'), 2)

            if "pump" in dn_type: q = pump_discharge(stage, on_elev, capacity)
            elif "gate" in dn_type or "spillway" in dn_type:
                q = gated_spillway_discharge(
                    headwater_elev=stage, tailwater_elev=0.8, crest_elev=crest_elev, 
                    gate_opening_height=1.0, spillway_width=width, 
                    design_head=design_head, num_piers=num_piers
                )
            q = max(0.0, q)
            discharge_per_node[f"{label} Discharge (cfs)"] = q
            total_q += q
        return total_q, discharge_per_node
        
    def _record_timestep_results(self, df_out, t, inflow, state, runoff_cfs, runoff_af, total_q, discharge_per_node):
        """Appends the results of a single timestep to the output dictionary."""
        df_out["Time (hr)"].append(t)
        df_out["Cumulative Rainfall (in)"].append(np.round(inflow.get('Cumulative Rainfall (in)', 0.0), 3))
        # ... (rest of the recording logic is unchanged)
        df_out["Incremental Rainfall (in)"].append(np.round(inflow.get('Incremental Rainfall (in)', 0.0), 3))
        df_out["Rainfall Excess (in)"].append(np.round(inflow.get('Rainfall Excess (in)', 0.0), 3))
        df_out["Runoff (cfs)"].append(np.round(runoff_cfs, 3))
        df_out["Runoff (acre-ft)"].append(runoff_af)
        df_out["Cumulative Runoff (acre-ft)"].append(np.round(state['cumulative_runoff'], 3))
        df_out["Stage (ft NGVD)"].append(np.round(state['stage'], 3))
        df_out["Storage (acre-ft)"].append(np.round(state['storage'], 3))
        df_out["Total Discharge (cfs)"].append(np.round(total_q, 3))
        df_out["Cumulative Total Discharge (acre-ft)"].append(np.round(state['cumulative_total_discharge'], 3))
        for col, val in discharge_per_node.items():
            df_out[col].append(np.round(val, 3))

    def _save_results(self):
        """Saves the simulation results to CSV files."""
        os.makedirs(self.outdir, exist_ok=True)
        for label, df_hydro in self.outlet_tables.items():
            safe_name = label.replace(" ", "_").replace("/", "_").replace("(", "").replace(")", "")
            df_hydro.to_csv(os.path.join(self.outdir, f"{safe_name}_hydrograph_full_results.csv"), index=False)
            
        summary_rows = [
            {"Node Label": label, "Max Stage (ft NGVD)": df["Stage (ft NGVD)"].max(), "Min Stage (ft NGVD)": df["Stage (ft NGVD)"].min()}
            for label, df in self.outlet_tables.items() if not df.empty
        ]
        pd.DataFrame(summary_rows).to_csv(os.path.join(self.outdir, "outlet_stage_summary.csv"), index=False)
        print(f"✅ Simulation completed. Results saved to: '{self.outdir}'")

        'Node Type': 'Pump', 'On Elev': 5.0, 'Capacity(cfs)': 50
    })
    G.add_edge("Basin1", "Pump1")
    
    # --- Instantiate and run the simulator with default time parameters (0-72 hrs, 1-hr interval) ---
